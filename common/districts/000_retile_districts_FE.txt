@city_cost = 500
@city_upkeep = 5

district_city = {
	base_buildtime = 480
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		exists = from
		from = { is_fallen_empire = yes }
		NOT = { is_planet_class = pc_habitat }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ringworld_habitable }
	}

	potential = {
		exists = owner
		owner = { is_fallen_empire = yes }
		NOT = { is_planet_class = pc_habitat }
		NOT = { is_planet_class = pc_city }
		NOT = { is_planet_class = pc_ringworld_habitable }
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = penal_colony }
	}
	
	allow = {
		NOT = { has_modifier = slave_colony }
		hidden_trigger = {
			retile_ready_city = yes
		}
	}

	conversion_ratio = 1
	convert_to = {
		district_clerks
	}
	
	resources = {
		category = planet_districts_cities
		cost = {
			alloys = @city_cost
		}
		upkeep = {
			trigger = { owner = { is_regular_empire = yes } }
			energy = @city_upkeep
		}
		upkeep = {
			trigger = { owner = { is_regular_empire = yes } }
			consumer_goods = 1
		}
	}
	
	triggered_planet_modifier = {						# Basic Jobs
		potential = {
			exists = owner
			owner = {
				is_regular_empire = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			planet_housing_add = 2
			job_researcher_add = 1
			job_foundry_add = 1
		}
	}
	triggered_planet_modifier = {						# Hive Jobs
		potential = {
			exists = owner
			owner = {
				is_hive_empire = yes
				is_fallen_empire = no
			}
		}
		modifier = {
			planet_housing_add = 2
			job_brain_drone_add = 1
			job_artisan_drone_add = 1
		}
	}
	triggered_planet_modifier = {						# Fallen Empire Housing
		potential = {
			exists = owner
			owner = {
				is_fallen_empire = yes
			}
		}
		modifier = {
			planet_housing_add = 4
		}
	}
	
	ai_weight = {
		weight = 100
		
		modifier = {	# Safe to build more cities?
			factor = 10
			retile_basic_economy = yes
		}
	}

	ai_resource_production = {
		alloys = 1
		consumer_goods = 1
		unity = 1
		physics_research = 1
		society_research = 1
		engineering_research = 1
		trigger = {
			always = yes
		}
	}
}
