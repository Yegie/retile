district_clerks = {
	base_buildtime = @retile_city_build_time
	icon = district_arcology_housing
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		exists = from
		from = { is_fallen_empire = no }
		#NOR = {
		#	is_planet_class = pc_habitat
		#	is_planet_class = pc_city
		#	is_planet_class = pc_ringworld_habitable
		#}
	}

	potential = {
		exists = owner
		owner = { is_fallen_empire = no }
		NOR = {
		#	is_planet_class = pc_habitat
		#	is_planet_class = pc_city
		#	is_planet_class = pc_ringworld_habitable
			has_modifier = resort_colony
			has_modifier = penal_colony
			has_modifier = slave_colony
		}
	}
	
	allow = {
		hidden_trigger = {
			retile_ready_city = yes
		}
	}

	resources = {
		category = planet_districts_cities
		cost = {
			alloys = @retile_city_cost
		}
		upkeep = {
			energy = @retile_city_upkeep
		}
	}
	
	triggered_planet_modifier = {						# Basic Jobs
		potential = {
			exists = owner
			NOR = {
				owner = { has_civic = civic_feudal_realm }
				owner = { has_civic = civic_capitalist }
			}
		}
		modifier = {
			planet_housing_add = 2
			job_clerk_add = 1
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_feudal_realm }
		}
		modifier = {
			planet_housing_add = 2
			job_tax_collector_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_civic = civic_capitalist }
		}
		modifier = {
			planet_housing_add = 2
			job_investor_add = 1
		}
	}

}
