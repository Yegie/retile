building_autochthon_monument = {
	base_buildtime = @retile_building_time_low
	base_cap_amount = 1
	category = unity
	can_be_disabled = no
	prerequisites = { tech_cultural_heritage }
	upgrades = { building_heritage_site }
	show_tech_unlock_if = {
		is_gestalt = no
		is_spiritualist = no
		is_megacorp = no
	}
	
	convert_to = {
		building_temple
		building_hive_node
		building_uplink_node
		building_corporate_monument
	}
	
	potential = {
		exists = owner
		owner = { 
			is_gestalt = no
			is_spiritualist = no
			is_megacorp = no
		}					
	}
	
	allow = {
		num_districts = { type = district_scholars value > 0 }
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}
	
	destroy_trigger = {
		OR = {
			num_districts = { type = district_scholars value < 1 }
			has_modifier = slave_colony
			has_modifier = penal_colony
			owner = { 
				OR = {
					is_gestalt = yes
					is_spiritualist = yes
					is_megacorp = yes
				}
			}
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			alloys = @retile_building_cost_low
		}
		upkeep = {
			energy = @retile_building_upkeep_nrg_low
		}
	}
	
	planet_modifier = {
		planet_housing_add = @retile_building_housing_low
	}
	
	country_modifier = {
		country_resource_max_add = @retile_building_storage_low
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_unity_1_effect
	}

	ai_weight = {
		weight = 1
	}

	ai_resource_production = {
		unity = 1
		trigger = {
			always = yes
		}
	}
}
