building_research_lab_1 = {
	base_buildtime = @retile_building_time_low
	base_cap_amount = 1
	category = research
	can_be_disabled = no
	upgrades = { building_research_lab_2 }
	prerequisites = { tech_basic_science_lab_1 }
	
	allow = {
		num_districts = { type = district_scholars value > 0 }
		NOT = { has_modifier = resort_colony }	
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}

	destroy_trigger = {
		OR = {
			num_districts = { type = district_scholars value < 1 }
			has_modifier = slave_colony
			has_modifier = resort_colony
			has_modifier = penal_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = @retile_building_cost_low
		}
		upkeep = {
			energy = @retile_building_upkeep_nrg_low
		}
	}
	
	planet_modifier = {
		planet_housing_add = @retile_building_housing_low
	}
	
	country_modifier = {
		country_resource_max_add = @retile_building_storage_low
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_lab_1_effect
	}

	ai_weight = {
		weight = 1
	}
	
	ai_resource_production = {
		physics_research = 1
		society_research = 1
		engineering_research = 1
		trigger = {
			always = yes
		}
	}
}

building_institute = {
	base_buildtime = @retile_building_time_med
	base_cap_amount = 1
	category = research
	can_be_disabled = no

	planet_modifier = {

	country_modifier = {
