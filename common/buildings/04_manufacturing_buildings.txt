building_foundry_1 = {
	base_buildtime = @retile_building_time_low
	base_cap_amount = 1
	category = manufacturing
	can_be_disabled = no
	upgrades = { building_foundry_2 }
	prerequisites = { tech_basic_industry }	
	
	allow = {
		num_districts = { type = district_engineers value > 0 }
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
	}
	
	destroy_trigger = {
		OR = {
			num_districts = { type = district_engineers value < 1 }
			has_modifier = resort_colony
			has_modifier = slave_colony
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			alloys = @retile_building_cost_low
		}
		upkeep = {
			energy = @retile_building_upkeep_nrg_low
		}
	}
	
	country_modifier = {
		country_resource_max_add = @retile_building_storage_low
	}
	
	planet_modifier = {
		planet_housing_add = @retile_building_housing_low
		#planet_pops_engineers_upkeep_mult = 1
		#planet_metallurgists_upkeep_mult = 1
		#planet_metallurgists_produces_mult = 1
	}
	
	custom_tooltip = retile_foundry_1_effect
	
	ai_weight = {
		weight = 1
	}

	ai_resource_production = {
		alloys = 1
		trigger = {
			always = yes
		}
	}
}
