building_hydroponics_farm = {
	base_buildtime = @retile_building_time_med
	base_cap_amount = 1
	category = resource
	can_be_disabled = no
	prerequisites = { tech_hydroponics }
	
	potential = {
		NOT = { has_modifier = slave_colony }
	}
	
	allow = {
		retile_ready_food = yes
	}
	
	destroy_trigger = {
		OR = {
			has_modifier = slave_colony
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			alloys = @retile_building_cost_med
		}
		upkeep = {
			energy = @retile_building_upkeep_nrg_med
		}
	}	
	
	planet_modifier = {
		planet_housing_add = @retile_building_housing_med
	}
	
	country_modifier = {
		country_resource_max_add = @retile_building_storage_med
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
		}
		text = retile_food_buff_city_effect
	}
	
	ai_weight = {
		weight = 0
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
