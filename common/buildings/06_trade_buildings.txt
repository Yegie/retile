building_marketplace = {
	base_buildtime = @retile_building_time_low
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	upgrades = { building_commercial_zone }
	
	potential = {
		exists = owner
		owner = { is_regular_empire = yes }
	}
	
	allow = {
		num_districts = { type = district_clerks value > 0 }
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
	}

	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			num_districts = { type = district_clerks value < 1 }
			has_modifier = slave_colony
			has_modifier = penal_colony
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			alloys = @retile_building_cost_low
			rare_crystals = @retile_building_cost_rare_low
		}
		upkeep = {
			energy = @retile_building_upkeep_nrg_low
		}
	}
	
	planet_modifier = {
		planet_housing_add = @retile_building_housing_low
	}
	
	country_modifier = {
		country_resource_max_add = @retile_building_storage_low
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_marketplace_effect
	}
	
	ai_weight = {
		weight = 1
	}
}
